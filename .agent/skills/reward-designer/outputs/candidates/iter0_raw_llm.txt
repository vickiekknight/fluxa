def reward_func_dist_to_target(env: ManagerBasedRLEnv, asset_cfg: SceneEntityCfg, command_name: str) -> torch.Tensor:
    """
    Reward for distance to target position.
    """
    robot = env.scene[asset_cfg.name]
    ee_pos = robot.data.body_pos_w[:, asset_cfg.body_ids[0]]
    target_pos = env.command_manager.get_command(command_name)[:, :3]
    distance = torch.norm(target_pos - ee_pos, dim=-1)
    # Reward is inversely proportional to distance, with a maximum reward for being within 5cm
    max_reward = 1.0
    min_dist_threshold = 0.05  # 5cm
    reward = torch.exp(-distance / min_dist_threshold) * max_reward
    return reward

def reward_func_orientation_mismatch(env: ManagerBasedRLEnv, asset_cfg: SceneEntityCfg, command_name: str) -> torch.Tensor:
    """
    Reward for orientation alignment.
    """
    robot = env.scene[asset_cfg.name]
    ee_orn_w = robot.data.body_orn_w[:, asset_cfg.body_ids[0]]
    target_orn = env.command_manager.get_command(command_name)[:, 3:] # Assuming orientation is next 4 values

    # Convert to axis-angle for easier comparison
    orientation_error = quat_diff_rad(ee_orn_w, target_orn)
    
    # Reward is higher when orientation error is smaller. Max reward for near perfect alignment.
    max_reward = 1.0
    max_orientation_error_rad = 0.1 # A small tolerance for orientation
    reward = torch.exp(-orientation_error / max_orientation_error_rad) * max_reward
    return reward

def reward_func_action_rate_penalty(env: ManagerBasedRLEnv, asset_cfg: SceneEntityCfg) -> torch.Tensor:
    """
    Penalty for high joint velocities.
    """
    robot = env.scene[asset_cfg.name]
    joint_vel = robot.data.joint_vel
    # Penalize large joint velocities to encourage smooth motion
    max_vel = 5.0 # Example maximum allowed joint velocity (rad/s)
    penalty = torch.sum(torch.square(joint_vel / max_vel), dim=-1)
    return -penalty

def reward_func_ee_vel_penalty(env: ManagerBasedRLEnv, asset_cfg: SceneEntityCfg) -> torch.Tensor:
    """
    Penalty for high end-effector linear velocity.
    """
    robot = env.scene[asset_cfg.name]
    ee_lin_vel_w = robot.data.body_lin_vel_w[:, asset_cfg.body_ids[0]]
    max_lin_vel = 1.0 # Example maximum allowed end-effector linear velocity (m/s)
    penalty = torch.sum(torch.square(ee_lin_vel_w / max_lin_vel), dim=-1)
    return -penalty
