def ee_position_reward(env: ManagerBasedRLEnv, asset_cfg: SceneEntityCfg, command_name: str) -> torch.Tensor:
    """
    Reward for reaching the target end-effector position.
    """
    robot = env.scene[asset_cfg.name]
    ee_pos = robot.data.body_pos_w[:, asset_cfg.body_ids[0]]
    target_pos = env.command_manager.get_command(command_name)[:, :3]
    distance = torch.norm(target_pos - ee_pos, dim=-1)
    # Reward is higher when distance is smaller. Scale is arbitrary but should be reasonable.
    # For 5cm tolerance, a distance of 0.05 should yield a reward of 0.
    reward = torch.exp(-5 * distance) * 10.0
    return reward

def ee_orientation_reward(env: ManagerBasedRLEnv, asset_cfg: SceneEntityCfg, command_name: str) -> torch.Tensor:
    """
    Reward for achieving the target end-effector orientation.
    """
    robot = env.scene[asset_cfg.name]
    ee_orn = robot.data.body_quat_w[:, asset_cfg.body_ids[0]]
    target_orn = env.command_manager.get_command(command_name)[:, 3:]

    # Calculate orientation difference (axis-angle representation)
    delta_orn = quat_mul(quat_inv(ee_orn), target_orn)
    # Angle is half the magnitude of the axis-angle vector.
    angle = torch.norm(delta_orn[:, 1:], dim=-1) * 2.0

    # Reward is higher when angle is smaller.
    reward = torch.exp(-5 * angle) * 10.0
    return reward

def action_rate_penalty(env: ManagerBasedRLEnv, asset_cfg: SceneEntityCfg, command_name: str) -> torch.Tensor:
    """
    Penalty for high joint velocities, encouraging smooth motion.
    """
    # Access the current and previous joint velocities
    joint_vel = env.scene[asset_cfg.name].data.joint_vel
    prev_joint_vel = env.scene[asset_cfg.name].data.prev_joint_vel
    
    # Calculate the difference in joint velocities
    vel_diff = joint_vel - prev_joint_vel
    
    # Square the difference and sum, then take the mean for each environment
    # A common approach is to penalize the L2 norm of the velocity changes.
    penalty = torch.mean(vel_diff**2, dim=-1)
    
    # The penalty should be a negative reward. Scale can be tuned.
    return -penalty * 0.1
