```python
# Domain randomization parameters for robot end-effector reaching task.
# The goal is to generalize the trained policy to the real world by simulating
# variations in physical properties of the robot and its environment.

# joint_friction_range:
# Friction in the joints can vary in the real world due to wear and tear,
# lubrication, and environmental factors. Randomizing friction can help the
# policy learn to be robust to these variations, ensuring smoother movements
# and avoiding unexpected behavior like oscillations or sluggishness.
# A moderate range is chosen to not make the task overly difficult to learn,
# but still capture significant variations.
joint_friction_range = [0.1, 2.0]

# joint_armature_range:
# Armature represents inertia associated with the joint itself, often due to
# gears or other transmission mechanisms. This can vary due to manufacturing
# tolerances or changes in the mechanical components over time.
# Randomizing armature helps the policy account for different effective
# inertias, leading to better control over accelerations and decelerations.
# A range that covers a reasonable variation without being extreme is selected.
joint_armature_range = [0.01, 0.5]

# joint_stiffness_scale_range:
# Actuator stiffness (Kp) determines how much force is applied in response
# to a position error. Variations in stiffness can arise from actuator
# degradation, different control gains in the real robot, or manufacturing
# variations. Randomizing stiffness helps the policy learn to adapt its
# control signals to different levels of joint responsiveness.
# A range around the default value is chosen to simulate typical variations.
joint_stiffness_scale_range = [0.8, 1.2]

# joint_damping_scale_range:
# Actuator damping (Kd) helps to dissipate energy and prevent oscillations.
# Similar to stiffness, damping can vary due to actuator properties or control
# parameters. Randomizing damping is crucial for ensuring smooth, jerk-free
# movements, as specified in the task. It helps the policy learn to compensate
# for different levels of damping, preventing overshooting or sluggish settling.
# A range around the default value is chosen to capture realistic variations
# that impact motion smoothness.
joint_damping_scale_range = [0.7, 1.3]
```
